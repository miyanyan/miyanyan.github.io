参考https://pip.pypa.io/en/stable/reference/requirement-specifiers/

#  System           platform value
# ---------------------------------
#  AIX              "aix"
#  Linux            "linux"
#  Windows          "win32"
#  Windows/Cygwin   "cygwin"
#  MacOS            "darwin"
# ---------------------------------
